\begin{exm} ({\bf ID 9915: Highest Cost Orders})\marginnote{This ID is from stratascratch.com}\\
Find the customers with the highest daily total order cost between 2019-02-01 and 2019-05-01. 
If a customer had more than one order on a certain day, sum the order costs on a daily basis. Output each customer's first name, total cost of their items, and the date. 
If multiple customers tie for the highest daily total on the same date, return all of them.

For simplicity, you can assume that every first name in the dataset is unique.

{\bf \color{red} Recall} the order of execution of statements in SQL~\cref{fig:SQL_LogicMargin}.
That means Window functions are evaluated after \code{WHERE}, so \code{max\_cost} doesn't exist yet when \code{WHERE} runs.
\begin{lstlisting}[language=SQL, linewidth=1.1\textwidth]
--- This will not work due to above remark.
WITH t as (SELECT customers.id, customers.first_name, 
                          orders.order_date, 
           SUM(orders.total_order_cost) as cust_spent_day
           FROM customers join orders on customers.id = orders.cust_id
           WHERE orders.order_date BETWEEN '2019-02-01' AND '2019-05-01'
           GROUP BY customers.id, customers.first_name, orders.order_date
           )
SELECT id, first_name, order_date, cust_spent_day,
MAX(cust_spent_day) over (partition by order_date) AS max_cost
from t
ORDER BY order_date, id
WHERE cust_spent_day = max_cost
\end{lstlisting}

\begin{lstlisting}[language=SQL]
-- This solution might not be optimal or readable or whatever
-- It shows what we could to to use the result of partition:
WITH daily_totals AS (SELECT c.id, c.first_name, o.order_date,
                      SUM(o.total_order_cost) AS cust_spent_day
                      FROM customers c
                     JOIN orders o ON c.id = o.cust_id
             WHERE o.order_date BETWEEN '2019-02-01' AND '2019-05-01'
                    GROUP BY c.id, c.first_name, o.order_date),
with_max AS (SELECT *,
      MAX(cust_spent_day) OVER (PARTITION BY order_date) AS max_cost
      FROM daily_totals)
SELECT first_name, order_date, max_cost
FROM with_max
WHERE cust_spent_day = max_cost
ORDER BY order_date, id;
\end{lstlisting}

\begin{lstlisting}[language=SQL]
-- Solution 2: no partition
WITH daily_totals AS (SELECT c.first_name, o.order_date,
                     SUM(o.total_order_cost) AS total_cost
              FROM customers c JOIN orders o ON c.id = o.cust_id
              WHERE o.order_date BETWEEN '2019-02-01' AND '2019-05-01'
                     GROUP BY c.first_name, o.order_date)
SELECT *
FROM daily_totals dt
WHERE total_cost = (SELECT MAX(total_cost)
                    FROM daily_totals
                    WHERE order_date = dt.order_date)
ORDER BY order_date, first_name;
\end{lstlisting}
\end{exm}

\begin{tcolorbox}[colback=aliceblue!10!white,colframe=blue!70!black,title=\textbf{Remark}]
% olback=green!10!white,colframe=green!80!black
\code{COALESCE} is a very handy SQL function that returns the first non-NULL value in its list of arguments:

\code{COALESCE(value1, value2, value3, \dots)}


\begin{lstlisting}[language=SQL]
SELECT name, COALESCE(middle_name, `N/A`) AS middle
FROM people;
\end{lstlisting}

So, in the example above:

If a company has no launches in 2020, \code{COUNT(DISTINCT \dots)} could be 
\vari{NULL} (depends on SQL dialect).
\code{COALESCE(\dots, 0)} ensures the result is zero instead of \vari{NULL} , so subtraction works safely.
\end{tcolorbox}

\begin{exm} ({\bf ID 10352: Users By Average Session Time})\\
Calculate each user's average session time, where a session is defined as the time difference between a  \code{page\_load} and a \code{page\_exit}. Assume each user has only one session per day. 
If there are multiple \code{page\_load} or \code{page\_exit} events on the same day, use only the latest 
 \code{page\_load} and the earliest  \code{page\_exit}.

\code{page\_exit} on the same day.

\begin{tcolorbox}[colback=aliceblue!10!white,colframe=blue!70!black,title=\textbf{Recall}]
I wasted so much time on this question:
In MySQL the correct and incorrect way of subtracting time stamps
seems to be in the following way
\begin{lstlisting}[language=SQL]
exit_.min_d_exit - load_.max_d_load -- incorrect
TIMESTAMPDIFF(SECOND, load_.max_d_load, exit_.min_d_exit) -- correct

-- JULIANDAY is for SQLite. In MySQL we have to use DATEDIFF
JULIANDAY(date2) - JULIANDAY(date1) >= 1
DATEDIFF(date2, date1) >= 1

-- JULIANDAY considers the year; the following will be 366
-- This is not to say DATEDIFF() does not do it! 
SELECT julianday('2025-01-01') - julianday('2024-01-01');
\end{lstlisting}
\end{tcolorbox}

\begin{lstlisting}[language=SQL, linewidth=1.2\textwidth]
WITH load_ AS (SELECT user_id, MAX(timestamp) AS max_d_load, 
               DATE(timestamp) AS date_
               FROM facebook_web_log
               WHERE action = 'page_load'
               GROUP BY user_id, DATE(timestamp)),
exit_ AS (SELECT user_id, MIN(timestamp) AS min_d_exit, 
          DATE(timestamp) AS date_
          FROM facebook_web_log
          WHERE action = 'page_exit'
          GROUP BY user_id, DATE(timestamp)
),
session_t AS (SELECT load_.user_id, 
        TIMESTAMPDIFF(SECOND, load_.max_d_load, exit_.min_d_exit) AS time_session 
           FROM load_ 
           JOIN exit_ USING (user_id, date_)
           WHERE exit_.min_d_exit > load_.max_d_load)
SELECT user_id, AVG(time_session)
FROM session_t
GROUP BY user_id;
\end{lstlisting}
\end{exm}

Calculate the friend acceptance rate for each \code{date} when friend requests were sent. 
A request is sent if \code{action = sent} and accepted if \code{action = accepted}. 
If a request is not accepted, there is no record of it being accepted in the table.

The output will only include dates where requests were sent and at least one of them was accepted (acceptance can occur on any date after the request is sent).

{\bf Shorter and clearer:}  Of all friend requests sent on a given day, what fraction eventually got accepted?
\marginnote{\idea Look at how left join is done by \code{a.date >= b.date}}

\begin{lstlisting}[language=SQL, linewidth=1.05\textwidth]
WITH a AS (SELECT * FROM fb_friend_requests WHERE action = 'accepted'),
     b AS (SELECT * FROM fb_friend_requests WHERE action = 'sent')
SELECT b.date AS send_date,
     ROUND(COUNT(a.user_id_receiver)*1.0/COUNT(*), 2) AS accept_rate
FROM b LEFT JOIN a ON b.user_id_sender = a.user_id_sender
                  AND b.user_id_receiver = a.user_id_receiver
                  AND a.date >= b.date
GROUP BY b.date
HAVING COUNT(a.user_id_receiver) > 0;
\end{lstlisting}
\end{exm}


\begin{exm} ({\bf \color{orange} ID 10322: Finding User Purchases})\\
Identify returning active users by finding users who made a second purchase within 1 to 7 days after their first purchase. Ignore same-day purchases. Output a list of these \code{user\_ids}.\\

\noindent {\color{red}{\bf NOTE}} Question is  talking only about 1st and 2nd purchase. Ignore 3rd, and 4th purchases.

\begin{lstlisting}[language=SQL, linewidth=1.3\textwidth]
with row_ as (SELECT *,
              ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY created_at ASC) as rn
              FROM amazon_transactions),
filtered as (SELECT *
             FROM row_
             WHERE rn <3),
help_t as (SELECT user_id,  created_at, 
  LAG (created_at) OVER (PARTITION BY user_id ORDER BY created_at ASC) as prev_dt_purch
           FROM filtered),
help_t2 as (SELECT *
            FROM help_t
            WHERE prev_dt_purch IS NOT NULL AND 
                      DATEDIFF(created_at, prev_dt_purch) >= 1 AND
                      DATEDIFF(created_at, prev_dt_purch) <= 7)
SELECT DISTINCT (user_id) 
FROM help_t2
\end{lstlisting}
\end{exm}

\begin{exm} ({\bf  \color{orange} ID 2081: Recommendation System})\\
You are given the list of Facebook friends and the list of Facebook pages that users follow. 
Your task is to create a new recommendation system for Facebook. 
For each Facebook user, find pages that this user doesn't follow but at least one of their friends does. 
Output the user ID and the ID of the page that should be recommended to this user.\sidenote{The following solutions are corrections of my solution by
chatGPT and Claude. i.e. I could not check against website's solution,
but hand investigation showed they are correct.}
\begin{lstlisting}[language=SQL, linewidth=1.05\textwidth]
WITH friend_pages AS (SELECT uf.user_id, up.page_id
                      FROM users_friends uf JOIN users_pages up
                              ON uf.friend_id = up.user_id)
SELECT DISTINCT fp.user_id, fp.page_id
FROM friend_pages fp
WHERE NOT EXISTS (SELECT 1
                  FROM users_pages up
              WHERE up.user_id = fp.user_id AND up.page_id = fp.page_id)
\end{lstlisting}

In the code above \code{WHERE NOT EXISTS} removes the pages
that a user already follows. 
\code{SELECT 1}: In SQL, \code{EXISTS} (or \code{NOT EXISTS}) 
does not care what you select — it only cares whether the subquery returns any rows.
\code{SELECT 1} can be replaced by \code{SELECT *} or  \code{SELECT 'abc'}.


Both above and following solutions are not
very intuitive, are they?
In the following, \code{WHERE up.page\_id IS NULL} keeps the
rows where no match was found; i.e. if user 1 and his friend user 2, both 
follow page A, then in the \code{fp} table, page A will show up for user 1 through user 2. 
But, we are looking for a page that user 1 is not already following. 
That is where \code{WHERE up.page\_id IS NULL} comes in
and kicks page A out of there.
\begin{lstlisting}[language=SQL, linewidth=1.21\textwidth]
WITH fp AS (SELECT uf.user_id, up.page_id
            FROM users_friends uf JOIN users_pages up ON uf.friend_id = up.user_id)
SELECT DISTINCT fp.user_id, fp.page_id
FROM fp LEFT JOIN users_pages up USING (user_id, page_id)
WHERE up.page_id IS NULL;
\end{lstlisting}

\noindent And here is another solution (still feels strange; does not come naturally yet)
\begin{lstlisting}[language=SQL, linewidth=1.05\textwidth]
SELECT DISTINCT uf.user_id, up.page_id
FROM users_friends uf JOIN users_pages up ON uf.friend_id = up.user_id
WHERE (uf.user_id, up.page_id) NOT IN (SELECT user_id, page_id 
                                       FROM users_pages)
ORDER BY uf.user_id, up.page_id;
\end{lstlisting}
\end{exm}


\begin{exm} ({\bf  \color{orange} ID 10300: Premium vs Freemium})\\
Find the total number of downloads for paying and non-paying users by date. Include only records where non-paying customers have more downloads than paying customers. The output should be sorted by earliest date first and contain 3 columns date, non-paying downloads, paying downloads. \\

\noindent Hint: In Oracle you should use "date" when referring to date column (reserved keyword).
Hint: The above hit is not a problem in MySQL. In fact, using quotation made a problem.
AND, in MySQL, there should not be a space in \code{MAX (}. It creates a problem and should be
\code{MAX(}.
\begin{lstlisting}[language=SQL, linewidth=1.5\textwidth]
with user_pay AS (SELECT * 
                  FROM ms_user_dimension LEFT JOIN ms_acc_dimension USING(acc_id)
                  ),
     all_data AS (SELECT *
                  FROM user_pay LEFT JOIN ms_download_facts USING (user_id)
                  ),
     counts AS (SELECT date, paying_customer, sum(downloads) as download_cnt
                FROM all_data
                GROUP BY date, paying_customer),
     pivoted AS (SELECT date,
                     MAX(CASE WHEN paying_customer = 'no' THEN download_cnt END) as count_non_paying,
                     MAX(CASE WHEN paying_customer = 'yes' THEN download_cnt END) as count_paying
                 FROM counts
                 GROUP BY date)
SELECT * 
FROM pivoted
WHERE count_non_paying > count_paying
ORDER BY date
\end{lstlisting}
\end{exm}
